package de.paladinsinn.tp.dcis.data.missions;

import com.vaadin.flow.data.provider.ListDataProvider;

import java.util.Arrays;

/**
 * SuccessState --
 *
 * @author klenkes74 {@literal <rlichti@kaiserpfalz-edv.de>}
 * @since 0.1.0  2021-03-28
 */
public enum SuccessState {
    FAILED,
    SUCCESS,
    GOOD,
    OUTSTANDING;

    public ListDataProvider<SuccessState> dataProvider() {
        return new ListDataProvider<>(Arrays.asList(FAILED, SUCCESS, GOOD, OUTSTANDING));
    }
}
