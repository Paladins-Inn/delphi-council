18  package de.paladinsinn.tp.dcis.ui.i18n;
19  
20  import lombok.extern.slf4j.Slf4j;
21  import org.jboss.logging.MDC;
22  import org.junit.jupiter.api.*;
23  
24  import java.util.Arrays;
25  import java.util.List;
26  import java.util.Locale;
27  import java.util.UUID;
28  
29  import static org.junit.jupiter.api.Assertions.assertArrayEquals;
30  import static org.junit.jupiter.api.Assertions.assertEquals;
31  
32  /**
33   * TranslatorTest -- Tests the common translator.
34   *
35   * @author klenkes74 {@literal <rlichti@kaiserpfalz-edv.de>}
36   * @since 0.1.0  2021-04-08
37   */
38  @Slf4j
39  class TranslatorTest {
40      private static final List<Locale> supportedLocales = Arrays.asList(Locale.GERMAN, Locale.ENGLISH);
41      private static final Locale DEFAULT_LOCALE = Locale.GERMAN;
42  
43  
44      /**
45       * Service under test.
46       */
47      private Translator sut;
48  
49      @Test
50      void shouldReturnListOfSupportedLocales() {
51          MDC.put("test-name", "supported-locales");
52  
53          List<Locale> result = sut.getProvidedLocales();
54          log.debug("result: {}", result.toArray());
55  
56          assertArrayEquals(supportedLocales.toArray(new Locale[]{}), result.toArray(new Locale[]{}), "The supported locals do not equal the expected ones.");
57      }
58  
59      @Test
60      void shouldReturnTheGermanTranslationWhenCalledWithAnExistingElementFromTheDefaultBundle() {
61          MDC.put("test-name", "valid-german-translation");
62  
63          String result = sut.getTranslation("existing.no-params", DEFAULT_LOCALE);
64          log.debug("result: {}", result);
65  
66          assertEquals("Dies ist ein Teststring.", result);
67      }
68  
69      @Test
70      void shouldReturnTheTranslationWhenCalledWithAnExistingElementFromTheClassLocalBundle() {
71          MDC.put("test-name", "valid-german-translation-class-bundle");
72  
73          String result = sut.getTranslation(this, "existing.no-params", DEFAULT_LOCALE);
74          log.debug("result: {}", result);
75  
76          assertEquals("Dies ist ein Teststring aus TranslatorTest.", result);
77      }
78  
79  
80      @Test
81      void shouldReturnAnErrorCodedStringWhenCalledWithAnNonExistingElementFromTheDefaultBundle() {
82          MDC.put("test-name", "missing-german-translation");
83  
84          String result = sut.getTranslation("non-existing", DEFAULT_LOCALE);
85          log.debug("result: {}", result);
86  
87          assertEquals("!non-existing", result);
88      }
89  
90      @Test
91      void shouldReturnTheStringWithParametersReplasedWhenCalledWithParamters() {
92          MDC.put("test-name", "adding-parameters-to-translation");
93  
94          UUID uuid = UUID.randomUUID();
95          String expected = String.format("Dies ist ein Teststring mit 1 Parameter: 0='%s'", uuid);
96  
97          String result = sut.getTranslation("existing.with-param", DEFAULT_LOCALE, uuid);
98          log.debug("result: {}", result);
99  
100         assertEquals(expected, result);
101     }
102 
103     @Test
104     void shouldReturnTheStringWithParametersReplasedWhenCalledWithTooFewParamters() {
105         MDC.put("test-name", "adding-too-few-parameters-to-translation");
106 
107         UUID uuid = UUID.randomUUID();
108         String expected = String.format("Dieser String hat 2 Parameter: 0='%s', 1='{1}'.", uuid);
109 
110         String result = sut.getTranslation("existing.with-params", DEFAULT_LOCALE, uuid);
111         log.debug("result: {}", result);
112 
113         assertEquals(expected, result);
114     }
115 
116     @Test
117     void shouldReturnTheStringWithTwoParametersReplasedWhenAllParametersAreGiven() {
118         MDC.put("test-name", "adding-two-parameters-to-translation");
119 
120         UUID uuid = UUID.randomUUID();
121         String param2 = "Und ein String";
122 
123         String expected = String.format("Dieser String hat 2 Parameter: 0='%s', 1='%s'.", uuid, param2);
124 
125         String result = sut.getTranslation("existing.with-params", DEFAULT_LOCALE, uuid, param2);
126         log.debug("result: {}", result);
127 
128         assertEquals(expected, result);
129     }
130 
131     @Test
132     void shouldReturnTheEnglishTranslationWhenCalledWithAnExistingElementFromTheDefaultBundle() {
133         MDC.put("test-name", "valid-english-translation");
134 
135         String result = sut.getTranslation("existing.no-params", Locale.ENGLISH);
136         log.debug("result: {}", result);
137 
138         assertEquals("This is a string to test.", result);
139     }
140 
141     @Test
142     void shouldFailWhenANonExistingMessageBundleIsSpecified() {
143         MDC.put("test-name", "non-existing-bundle");
144 
145         String result = sut.getTranslation("ignored", "ignored", Locale.GERMAN);
146         log.debug("result: {}", result);
147 
148         assertEquals("!ignored", result);
149     }
150 
151 
152     @Test
153     void close() throws Exception {
154         sut.close();
155     }
156 
157 
158     @BeforeEach
159     void setUp() {
160         sut = new Translator("test-messages");
161     }
162 
163     @AfterEach
164     void tearDown() throws Exception {
165         sut.close();
166         MDC.remove("test");
167     }
168 
169     @BeforeAll
170     static void setUpAll() {
171         MDC.put("test-class", TranslatorTest.class.getSimpleName());
172         log.info("Starting tests.");
173     }
174 
175     @AfterAll
176     static void tearDownAll() {
177         log.info("Tests ended.");
178         MDC.remove("test-class");
179     }
180 }
