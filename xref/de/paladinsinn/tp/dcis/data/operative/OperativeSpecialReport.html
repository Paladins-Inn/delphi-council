18  package de.paladinsinn.tp.dcis.data.operative;
19  
20  import de.paladinsinn.tp.dcis.data.AbstractRevisionedEntity;
21  import de.paladinsinn.tp.dcis.data.HasName;
22  import de.paladinsinn.tp.dcis.data.specialmissions.SpecialMission;
23  import lombok.Getter;
24  import lombok.Setter;
25  import org.apache.commons.lang3.builder.CompareToBuilder;
26  import org.hibernate.envers.Audited;
27  
28  import javax.persistence.*;
29  import java.util.StringJoiner;
30  
31  /**
32   * OperativeSpecialReport -- The special report for a local table game.
33   *
34   * @author klenkes74 {@literal <rlichti@kaiserpfalz-edv.de>}
35   * @since 0.3.0  2021-04-18
36   */
37  @Entity
38  @Audited
39  @Table(
40          name = "SPECIALREPORTS_OPERATIVES",
41          uniqueConstraints = {
42                  @UniqueConstraint(name = "SPECIALREPORTS_OPERATIVES_UK", columnNames = {"SPECIALMISSION_ID", "OPERATIVE_ID"}),
43          }
44  )
45  @Getter
46  @Setter
47  public class OperativeSpecialReport extends AbstractRevisionedEntity implements Comparable<OperativeSpecialReport>, HasName, HasOperative {
48      @Column(name = "NOTES", length = 4000)
49      private String notes;
50  
51      @ManyToOne(
52              targetEntity = SpecialMission.class,
53              fetch = FetchType.EAGER,
54              cascade = {CascadeType.REFRESH},
55              optional = false
56      )
57      @JoinColumn(name = "SPECIALMISSION_ID", nullable = false)
58      private SpecialMission specialMission;
59  
60      @ManyToOne(
61              targetEntity = Operative.class,
62              fetch = FetchType.EAGER,
63              cascade = {CascadeType.REFRESH},
64              optional = false
65      )
66      @JoinColumn(name = "OPERATIVE_ID", nullable = false)
67      private Operative operative;
68  
69  
70      @Transient
71      public void setSpecialMission(SpecialMission specialMission) {
72          SpecialMission old = this.specialMission;
73          this.specialMission = specialMission;
74  
75          if (specialMission != null && !specialMission.equals(old)) {
76              specialMission.addOperativeReport(this);
77  
78              if (old != null) {
79                  old.removeOperativeReport(this);
80              }
81          }
82      }
83  
84      @Override
85      @Transient
86      public void setOperative(Operative operative) {
87          Operative old = this.operative;
88          this.operative = operative;
89  
90          if (operative != null && !operative.equals(old)) {
91              operative.addSpecialReport(this);
92  
93              if (old != null) {
94                  old.removeSpecialReport(this);
95              }
96          }
97      }
98  
99      @Override
100     @Transient
101     public String getName() {
102         return operative.getName();
103     }
104 
105     @Override
106     @Transient
107     public void setName(final String name) {
108     }
109 
110 
111     @Override
112     public int compareTo(OperativeSpecialReport o) {
113         return new CompareToBuilder()
114                 .append(getSpecialMission(), o.getSpecialMission())
115                 .append(getOperative(), o.getOperative())
116                 .toComparison();
117     }
118 
119 
120     @Override
121     public OperativeSpecialReport clone() throws CloneNotSupportedException {
122         OperativeSpecialReport result = (OperativeSpecialReport) super.clone();
123 
124         result.operative = operative.clone();
125         result.specialMission = specialMission.clone();
126 
127         result.notes = notes;
128 
129         return result;
130     }
131 
132 
133     @Override
134     public String toString() {
135         return new StringJoiner(", ", OperativeSpecialReport.class.getSimpleName() + "[", "]")
136                 .merge(super.getToStringJoiner())
137 
138                 .add("report=" + specialMission.getId())
139                 .add("operative='" + operative.getName() + "'")
140 
141                 .toString();
142     }
143 }
