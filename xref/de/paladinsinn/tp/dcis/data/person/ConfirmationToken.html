18  package de.paladinsinn.tp.dcis.data.person;
19  
20  import com.sun.istack.NotNull;
21  import de.paladinsinn.tp.dcis.data.AbstractEntity;
22  import lombok.Getter;
23  import lombok.NoArgsConstructor;
24  import lombok.Setter;
25  import org.slf4j.Logger;
26  import org.slf4j.LoggerFactory;
27  
28  import javax.persistence.*;
29  import java.util.StringJoiner;
30  
31  /**
32   * ConfirmationToken --
33   *
34   * @author Kamer Elciar (<a href="https://kamer-dev.medium.com/spring-boot-user-registration-and-login-43a33ea19745">https://kamer-dev.medium.com/spring-boot-user-registration-and-login-43a33ea19745</a>)
35   * @author klenkes74 {@literal <rlichti@kaiserpfalz-edv.de>}
36   * @since 0.1.0  2021-04-10
37   */
38  @Entity(name = "token")
39  @Table(name = "CONFIRMATIONTOKENS")
40  @Getter
41  @Setter
42  @NoArgsConstructor
43  public class ConfirmationToken extends AbstractEntity {
44      private static final Logger LOG = LoggerFactory.getLogger(ConfirmationToken.class);
45  
46      @OneToOne(
47              targetEntity = Person.class,
48              fetch = FetchType.EAGER
49      )
50      @JoinColumn(name = "PERSON_ID", nullable = false)
51      private Person person;
52  
53      /**
54       * Creates a ConfirmationToken for the given person.
55       *
56       * @param person the person the confirmation token is generated for.
57       */
58      public ConfirmationToken(@NotNull final Person person) {
59          this.person = person;
60  
61          prePersist();
62      }
63  
64      @Override
65      public String toString() {
66          return new StringJoiner(", ", ConfirmationToken.class.getSimpleName() + "[", "]")
67                  .merge(super.getToStringJoiner())
68                  .add("person=" + person)
69                  .toString();
70      }
71  }
