256     /**
257      * Checks the password handed in.
258      *
259      * @param pw The password to check.
260      * @return TRUE if the passwords match.
261      */
262     @SuppressWarnings("unused")
263     public boolean checkPassword(final String pw) {
264         boolean result = new BCryptPasswordEncoder().matches(pw, password);
265 
266         if (result) status.setLastLogin(getNowUTC());
267         return result;
268     }
269 
270     /**
271      * Updates the password of the user.
272      *
273      * @param pw The new password.
274      */
275     @SuppressWarnings("unused")
276     public void setPassword(final String pw) {
277         password = new BCryptPasswordEncoder().encode(pw);
278 
279         preUpdate();
280         status.setCredentialsChange(getModified());
281     }

386     @Override
387     public Person clone() throws CloneNotSupportedException {
388         Person result = (Person) super.clone();
389 
390         result.name = name;
391         result.username = username;
392         result.firstname = firstname;
393         result.lastname = lastname;
394         result.email = email;
395         result.password = password;
396         result.locale = locale;
397 
398         result.roles = Set.copyOf(roles);
399         result.operatives = Set.copyOf(operatives);
400         result.reports = Set.copyOf(reports);
401 
402         result.status = status.clone();
403         result.avatar = avatar.clone();
404 
405         return result;
406     }
