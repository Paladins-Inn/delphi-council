101     @PostConstruct

103         setLastLoggedIn();

105         menu = createMenu();

108         addToNavbar(true, createHeaderContent());
109         addToDrawer(createDrawerContent(menu));

112     private void setLastLoggedIn() {
113         if (user != null) {
114             LOG.debug("Setting last login date to user. user='{}'", user.getPerson().getUsername());
115 
116             user.getPerson().getStatus().setLastLogin(OffsetDateTime.now(ZoneOffset.UTC));
117             user.setPerson(personRepository.save(user.getPerson()));
118 
119             locale = user.getPerson().getLocale();
120             LOG.debug("Setting locale to user locale. locale={}", locale);
121         } else {
122             LOG.info("No user logged in.");
123             locale = VaadinSession.getCurrent().getLocale();
124         }
125     }

127     private Component createHeaderContent() {
128         HorizontalLayout layout = new HorizontalLayout();
129         layout.setId("header");
130         layout.getThemeList().set("dark", true);
131         layout.addClassName("torg-header");
132         layout.setWidthFull();
133         layout.setSpacing(false);
134         layout.setAlignItems(FlexComponent.Alignment.CENTER);
135         layout.add(new DrawerToggle());
136         titleBar = new H1();
137         layout.add(titleBar);
138 
139         Authentication authentication = SecurityContextHolder.getContext().getAuthentication();
140 
141         Avatar avatar = createAvatar(authentication);
142         layout.add(avatar);
143 
144         return layout;
145     }

147     private Avatar createAvatar(Authentication authentication) {
148         Avatar result = new Avatar(authentication.getName());
149 
150         Person data = personRepository.findByUsername(authentication.getName());
151         result.setImage(data.getGravatar());
152 
153         ContextMenu menu = new ContextMenu();
154         menu.setTarget(result);
155 
156         menu.addItem(getTranslation("avatar.menu.account-settings"),
157                 e -> {
158                     LOG.info("User opens settings menu. event={}", e);
159 
160                     e.getSource().getUI().ifPresent(ui -> ui.navigate(
161                             PersonEditView.class,
162                             new RouteParameters("id", user.getPerson().getId().toString())
163                     ));
164                 }
165         );
166 
167         menu.addItem(getTranslation("buttons.logout.caption"),
168                 e -> {
169                     LOG.info("User wanted to log out. event={}", e);
170                     e.getSource().getUI().ifPresent(ui -> ui.getPage().open("/logout", "_self"));
171                 }
172         );
173 
174 
175         return result;
176     }

206     private Tabs createMenu() {
207         final Tabs tabs = new Tabs();
208         tabs.setOrientation(Tabs.Orientation.VERTICAL);
209         tabs.addThemeVariants(TabsVariant.LUMO_MINIMAL);
210         tabs.setId("tabs");
211         tabs.add(createMenuItems());
212         return tabs;
213     }
214 
215     private Component[] createMenuItems() {
216         ArrayList<Component> result = new ArrayList<>();
217 
218         ArrayList<Class<? extends Component>> allTabs = new ArrayList<>();
219         allTabs.add(MissionListView.class);
220         allTabs.add(SpecialMissionListView.class);
221         allTabs.add(SpecialMissionEditorView.class);
222         allTabs.add(OperativesListView.class);
223         allTabs.add(PersonListView.class);
224         allTabs.add(ThreatCardView.class);
225         allTabs.add(AboutView.class);
226 
227         Set<String> roles = readRoleFromAuthentication();
228 
229         for (Class<? extends Component> t : allTabs) {
230             Secured secured = t.getAnnotation(Secured.class);
231 
232             if (secured == null || Arrays.stream(secured.value()).anyMatch(roles::contains)) {
233                 LOG.trace(
234                         "View is either free or role matches. view={}, roles={}, rolesAllowed={}",
235                         t.getSimpleName(),
236                         roles,
237                         secured != null ? Arrays.asList(secured.value()) : "-not-set-"
238                 );
239 
240 
241                 Tab tab = createTab(getTranslation(t.getSimpleName()), t);
242                 result.add(tab);
243             } else {
244                 LOG.debug("View is not allowed for user. view={}, roles={}, rolesAllowed={}",
245                         t.getSimpleName(),
246                         roles,
247                         Arrays.asList(secured.value())
248                 );
249             }
250         }
251 
252         return result.toArray(new Component[]{});
253     }
254 
255     private Set<String> readRoleFromAuthentication() {
256         HashSet<String> result = new HashSet<>(user.getPerson().getAuthorities().size());
257 
258         user.getPerson().getAuthorities().forEach(r -> result.add(r.getAuthority()));
259 
260         return result;
261     }
262 
263     private static Tab createTab(String text, Class<? extends Component> navigationTarget) {
264         final Tab tab = new Tab();
265         tab.add(new RouterLink(text, navigationTarget));
266         ComponentUtil.setData(tab, Class.class, navigationTarget);
267         return tab;
268     }

270     @Override
271     protected void afterNavigation() {
272         super.afterNavigation();
273         getTabForComponent(getContent()).ifPresent(menu::setSelectedTab);
274         titleBar.setText(menu.getSelectedTab().getLabel());
275 //        titleBar.setText(getCurrentPageTitle());
276     }
277 
278     private Optional<Tab> getTabForComponent(Component component) {
279         return menu.getChildren().filter(tab -> ComponentUtil.getData(tab, Class.class).equals(component.getClass()))
280                 .findFirst().map(Tab.class::cast);
281     }

283     @Override
284     public void localeChange(@NotNull final LocaleChangeEvent event) {
285         LOG.trace("Language change event. locale={}", event.getLocale());
286 
287         Locale locale = event.getLocale();
288 
289         setLocale(locale);
290     }
291 
292     public void setLocale(@NotNull final Locale locale) {
293         if (this.locale != null && this.locale.equals(locale)) {
294             LOG.trace("Language did not change - ignoring event. locale={}", this.locale);
295             return;
296         }
297 
298         this.locale = locale;
299 
300         translate();
301     }
