18  package de.paladinsinn.tp.dcis.ui.views.missions;
19  
20  import com.sun.istack.NotNull;
21  import com.vaadin.flow.component.dependency.CssImport;
22  import com.vaadin.flow.component.html.Div;
23  import com.vaadin.flow.i18n.LocaleChangeEvent;
24  import com.vaadin.flow.i18n.LocaleChangeObserver;
25  import com.vaadin.flow.router.BeforeEnterEvent;
26  import com.vaadin.flow.router.BeforeEnterObserver;
27  import com.vaadin.flow.router.Route;
28  import com.vaadin.flow.server.VaadinSession;
29  import de.paladinsinn.tp.dcis.data.missions.Mission;
30  import de.paladinsinn.tp.dcis.data.missions.MissionRepository;
31  import de.paladinsinn.tp.dcis.ui.MainView;
32  import de.paladinsinn.tp.dcis.ui.i18n.I18nPageTitle;
33  import de.paladinsinn.tp.dcis.ui.i18n.TranslatableComponent;
34  import org.slf4j.Logger;
35  import org.slf4j.LoggerFactory;
36  import org.springframework.beans.factory.annotation.Autowired;
37  import org.springframework.security.access.annotation.Secured;
38  
39  import javax.annotation.PostConstruct;
40  import java.io.Serializable;
41  import java.util.Locale;
42  import java.util.Optional;
43  import java.util.UUID;
44  
45  /**
46   * MissionEditorView -- Editor for adding new {@link Mission}.
47   *
48   * @author klenkes74 {@literal <rlichti@kaiserpfalz-edv.de>}
49   * @since 0.1.0  2021-04-06
50   */
51  @Route(value = "mission/:id?", layout = MainView.class)
52  @I18nPageTitle("mission.editor.caption")
53  @CssImport("./views/edit-view.css")
54  @Secured({"JUDGE", "ORGA", "ADMIN"})
55  public class MissionEditorView extends Div implements BeforeEnterObserver, LocaleChangeObserver, TranslatableComponent, Serializable, AutoCloseable {
56      private static final Logger LOG = LoggerFactory.getLogger(MissionEditorView.class);
57  
58      @Autowired
59      private MissionForm missionForm;
60  
61      @Autowired
62      private MissionRepository missionRepository;
63  
64      private Locale locale;
65  
66      @PostConstruct
67      public void init() {
68          setLocale(VaadinSession.getCurrent().getLocale());
69          LOG.debug("Loading form. locale={}", locale);
70  
71          addClassName("edit-view");
72  
73          add(missionForm);
74          translate();
75      }
76  
77  
78      @Override
79      public void beforeEnter(BeforeEnterEvent event) {
80          Optional<String> id = event.getRouteParameters().get("id");
81          LOG.trace("Entering form. id={}", id);
82  
83          id.ifPresentOrElse(
84                  e -> missionRepository.findById(UUID.fromString(e)).ifPresentOrElse(
85                          this::setData,
86                          () -> setData(new Mission())
87                  ),
88                  () -> setData(new Mission())
89          );
90      }
91  
92      private void setData(@NotNull Mission data) {
93          if (data == null || data.equals(missionForm.getData().orElse(null))) {
94              LOG.info("Data is null or data didn't change - will do nothing. data={}", data);
95              return;
96          }
97          LOG.trace("Loaded mission. mission={}", data);
98  
99          missionForm.setData(data);
100         translate();
101     }
102 
103 
104     @Override
105     public void localeChange(LocaleChangeEvent event) {
106         LOG.trace("Locale change event. locale={}", event.getLocale());
107 
108         setLocale(event.getLocale());
109     }
110 
111     @Override
112     public void translate() {
113         if (locale == null || missionForm.getData().isEmpty()) {
114             LOG.debug("No locale or no data - will do nothing. locale={}, data={}", locale, missionForm.getData());
115             return;
116         }
117 
118         missionForm.translate();
119     }
120 
121     @Override
122     public void setLocale(Locale locale) {
123         if (this.locale != null && this.locale.equals(locale)) {
124             LOG.debug("Locale didn't change - will do nothing. locale={}", locale);
125             return;
126         }
127 
128         this.locale = locale;
129 
130         missionForm.setLocale(locale);
131         translate();
132     }
133 
134     @Override
135     public void close() throws Exception {
136         LOG.trace("Closing form.");
137 
138         missionForm.close();
139 
140         removeAll();
141 
142         LOG.debug("Closed form.");
143     }
144 }
