50  /**
51   * MissionEditForm -- Edits/displays the data for a mission.
52   *
53   * @author klenkes74 {@literal <rlichti@kaiserpfalz-edv.de>}
54   * @since 0.1.0  2021-03-29

56  @Service
57  @Scope(SCOPE_PROTOTYPE)
58  public class MissionForm extends Composite<Div> implements LocaleChangeObserver, TranslatableComponent, Serializable, AutoCloseable {
59      private static final Logger LOG = LoggerFactory.getLogger(MissionForm.class);
60      public static final int DEFAULT_PAYMENT = 250;
61      public static final int DEFAULT_XP = 5;
62  
63      /**
64       * service for writing data to.
65       */
66      private final MissionService missionService;
67  
68      /**
69       * The locale of the form.
70       */
71      private Locale locale;
72  
73      private final LoggedInUser user;
74  
75      private final TextField title = new TextField();
76      private final FormLayout form = new FormLayout();
77  
78      private final TextField id = new TextField();
79      private final TextField code = new TextField();
80      private final ComboBox<Clearance> clearance = new ComboBox<>();
81      private final TextField publication = new TextField();
82      private final TextField image = new TextField();
83      private final TextArea description = new TextArea();
84      private final IntegerField payment = new IntegerField();
85      private final IntegerField xp = new IntegerField();
86      private final TextArea objectivesSuccess = new TextArea();
87      private final TextArea objectivesGood = new TextArea();
88      private final TextArea objectivesOutstanding = new TextArea();
89  
90      private TorgActionBar actions;
91  
92      /**
93       * mission data
94       */
95      private Mission data;
96  
97      /**
98       * If the form is already initialized.
99       */
100     private boolean initialized = false;
101 
102     @Autowired
103     public MissionForm(
104             @NotNull final MissionService missionService,
105             @NotNull final LoggedInUser user
106     ) {
107         this.missionService = missionService;
108         this.user = user;
109     }
110 
111     private void init() {
112         if (data == null || user == null || initialized) {
113             LOG.debug(
114                     "Can't initialize or already initialized. initialized={}, data={}, user={}",
115                     initialized, data, user
116             );
117             return;
118         }
119 
120         if (locale == null) {
121             locale = VaadinSession.getCurrent().getLocale();
122         }
123 
124         addListener(MissionSaveEvent.class, missionService);
125 
126         form.setResponsiveSteps(
127                 new FormLayout.ResponsiveStep("1px", 1),
128                 new FormLayout.ResponsiveStep("400px", 2),
129                 new FormLayout.ResponsiveStep("800px", 3)
130         );
131 
132         actions = new TorgActionBar(
133                 "buttons",
134                 event -> {
135                     scrape();
136                     getEventBus().fireEvent(new MissionSaveEvent(this, false));
137                 },
138                 null,
139                 null,
140                 null
141         );
142         actions.setReadOnly(
143                 user.isReadonly()
144                         && !user.isAdmin()
145                         && !user.isOrga()
146                         && !user.isJudge()
147         );
148 
149 
150         id.setReadOnly(true);
151 
152         code.setRequired(true);
153         code.setReadOnly(user.isReadonly());
154 
155         title.setRequired(true);
156         title.setReadOnly(user.isReadonly());
157 
158         clearance.setDataProvider(Clearance.ANY.dataProvider());
159         clearance.setAllowCustomValue(false);
160         clearance.setRequired(true);
161         clearance.setReadOnly(user.isReadonly());
162 
163         publication.setRequired(false);
164         publication.setReadOnly(user.isReadonly());
165 
166         image.setRequired(false);
167         image.setReadOnly(user.isReadonly());
168 
169         description.addClassName("long-text");
170         description.setRequired(true);
171         description.setReadOnly(user.isReadonly());
172 
173         payment.setMin(100);
174         payment.setMax(1000);
175         payment.setStep(50);
176         payment.setValue(DEFAULT_PAYMENT);
177         payment.setReadOnly(user.isReadonly());
178 
179         xp.setMin(1);
180         xp.setMax(50);
181         xp.setStep(1);
182         xp.setValue(DEFAULT_XP);
183         xp.setReadOnly(user.isReadonly());
184 
185         objectivesSuccess.addClassName("long-text");
186         objectivesSuccess.setRequired(true);
187         objectivesSuccess.setReadOnly(user.isReadonly());
188 
189         objectivesGood.addClassName("long-text");
190         objectivesGood.setRequired(true);
191         objectivesGood.setReadOnly(user.isReadonly());
192 
193         objectivesOutstanding.addClassName("long-text");
194         objectivesOutstanding.setRequired(true);
195         objectivesOutstanding.setReadOnly(user.isReadonly());
196 
197         getContent().add(form);
198 
199         initialized = true;
200     }
201 
202     public void setData(@NotNull Mission data) {
203         if (this.data != null && this.data.equals(data)) {
204             LOG.info("Mission report didn't change. Ignoring event. code={}, id={}",
205                     this.data.getCode(), this.data.getId());
206 
207             return;
208         }
209 
210         LOG.debug("Set mission. id={}, code={}, title={}", data.getId(), data.getCode(), data.getName());
211 
212         this.data = data;
213 
214         init();
215         populate();
216         translate();
217     }
218 
219     public void populate() {
220         if (data == null) {
221             LOG.warn("Tried to populate form data without a mission defined.");
222             return;
223         }
224 
225         if (data.getId() != null) {
226             id.setValue(data.getId().toString());
227         }
228 
229         bindData(data.getCode(), code);
230         bindData(data.getName(), title);
231         bindData(data.getClearance(), clearance);
232         bindData(data.getPublication(), publication);
233         bindData(data.getImage(), image);
234         bindData(data.getDescription(), description);
235         bindData(data.getPayment(), payment);
236         bindData(data.getXp(), xp);
237         bindData(data.getObjectivesSuccess(), objectivesSuccess);
238         bindData(data.getObjectivesGood(), objectivesGood);
239         bindData(data.getObjectivesOutstanding(), objectivesOutstanding);
240     }
241 
242     @SuppressWarnings({"rawtypes", "unchecked"})
243     private void bindData(final Object data, @NotNull HasValue component) {
244         if (data != null) {
245             component.setValue(data);
246         }
247     }
248 
249     private void scrape() {
250         if (data == null) data = new Mission();
251 
252         data.setCode(code.getValue());
253         data.setName(title.getValue());
254         data.setClearance(clearance.getValue());
255         data.setPublication(publication.getValue());
256         data.setImage(image.getValue());
257         data.setDescription(description.getValue());
258         data.setPayment(payment.getValue());
259         data.setXp(xp.getValue());
260         data.setObjectivesSuccess(objectivesSuccess.getValue());
261         data.setObjectivesGood(objectivesGood.getValue());
262         data.setObjectivesOutstanding(objectivesOutstanding.getValue());
263         data.preUpdate();
264     }
265 
266 
267     public Optional<Mission> getData() {
268         return Optional.ofNullable(data);
269     }

273         LOG.trace("Change locale event. locale={}", event.getLocale());

278     @Override
279     public void translate() {
280         if (user == null || data == null || locale == null) {
281             LOG.warn(
282                     "Can't build mission form. user={}, mission={}, locale={}",
283                     user, data, locale
284             );
285 
286             return;
287         }
288 
289         LOG.debug("Building mission edit form. locale={}, mission={}", locale, data);
290 
291         LOG.trace("Removing all form elements.");
292         form.removeAll();
293 
294 
295         // Form fields
296         LOG.trace("Translating form elements.");
297         id.setLabel(getTranslation("input.id.caption"));
298         id.setHelperText(getTranslation("input.id.help"));
299 
300         code.setLabel(getTranslation("mission.code.caption"));
301         code.setHelperText(getTranslation("mission.code.help"));
302 
303         title.setLabel(getTranslation("mission.title.caption"));
304         title.setHelperText(getTranslation("mission.title.help"));
305 
306         clearance.setLabel(getTranslation("torg.clearance.caption"));
307         clearance.setHelperText(getTranslation("torg.clearance.help"));
308         clearance.setItemLabelGenerator((ItemLabelGenerator<Clearance>) item -> getTranslation("torg.clearance." + item.name()));
309 
310         publication.setLabel(getTranslation("mission.publication.caption"));
311         publication.setHelperText(getTranslation("mission.publication.help"));
312 
313         image.setLabel(getTranslation("mission.image.caption"));
314         image.setHelperText(getTranslation("mission.image.help"));
315 
316         description.setLabel(getTranslation("mission.description.caption"));
317         description.setHelperText(getTranslation("mission.description.help"));
318 
319         payment.setLabel(getTranslation("mission.payment.caption"));
320         payment.setHelperText(getTranslation("mission.description.help"));
321 
322         xp.setLabel(getTranslation("mission.xp.caption"));
323         xp.setHeight(getTranslation("mission.xp.help"));
324 
325         objectivesSuccess.setLabel(getTranslation("mission.objectives.success.caption"));
326         objectivesSuccess.setHelperText(getTranslation("mission.objectives.success.help"));
327 
328         objectivesGood.setLabel(getTranslation("mission.objectives.good.caption"));
329         objectivesGood.setHelperText(getTranslation("mission.objectives.good.help"));
330 
331         objectivesOutstanding.setLabel(getTranslation("mission.objectives.outstanding.caption"));
332         objectivesOutstanding.setHelperText(getTranslation("mission.objectives.outstanding.help"));
333 
334 
335         LOG.trace("adding all form elements.");
336         form.add(code, clearance, title, publication, payment, xp, image, description,
337                 objectivesSuccess, objectivesGood, objectivesOutstanding);
338 
339         form.add(actions);
340         form.setColspan(actions, 3);
341 
342 
343         form.setColspan(title, 3);
344         form.setColspan(publication, 3);
345         form.setColspan(description, 3);
346         form.setColspan(objectivesSuccess, 3);
347         form.setColspan(objectivesGood, 3);
348         form.setColspan(objectivesOutstanding, 3);
349     }

352     @Override
353     public void setLocale(Locale locale) {
354         if (this.locale != null && this.locale.equals(locale)) {
355             LOG.debug("Locale has not changed. Ignoring event. locale={}", locale);
356             return;
357         }
358 
359         this.locale = locale;
360         translate();
361     }
362 
363     private String getTranslation(@NotNull final String key) {
364         try {
365             return super.getTranslation(key);
366         } catch (NullPointerException e) {
367             LOG.warn("Can't call translator from vaadin: {}", e.getMessage());
368             return "!" + key;
369         }
370     }
371 
372     @Override
373     public void close() throws Exception {
374         LOG.debug("Closing form.");
375         getContent().removeAll();
376         form.removeAll();
377     }
