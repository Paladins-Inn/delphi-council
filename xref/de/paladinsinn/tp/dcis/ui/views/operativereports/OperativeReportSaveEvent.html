18  package de.paladinsinn.tp.dcis.ui.views.operativereports;
19  
20  import com.vaadin.flow.component.ComponentEvent;
21  import de.paladinsinn.tp.dcis.data.operative.OperativeReport;
22  
23  /**
24   * MissionGrouPReportSaveEvent --
25   *
26   * @author klenkes74 {@literal <rlichti@kaiserpfalz-edv.de>}
27   * @since 0.1.0  2021-04-04
28   */
29  public class OperativeReportSaveEvent extends ComponentEvent<OperativeReportForm> {
30      /**
31       * Creates a new event using the given source and indicator whether the
32       * event originated from the client side or the server side.
33       *
34       * @param source     the source component
35       * @param fromClient <code>true</code> if the event originated from the client
36       */
37      public OperativeReportSaveEvent(OperativeReportForm source, boolean fromClient) {
38          super(source, fromClient);
39      }
40  
41      /**
42       * @return the mission report to be saved.
43       */
44      public OperativeReport getReport() {
45          return getSource().getData().orElseThrow();
46      }
47  }
