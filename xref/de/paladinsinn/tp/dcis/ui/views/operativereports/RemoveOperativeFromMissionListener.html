18  package de.paladinsinn.tp.dcis.ui.views.operativereports;
19  
20  import com.vaadin.flow.component.ComponentEventListener;
21  import de.paladinsinn.tp.dcis.data.missions.MissionReport;
22  import de.paladinsinn.tp.dcis.data.missions.MissionReportRepository;
23  import de.paladinsinn.tp.dcis.data.operative.Operative;
24  import de.paladinsinn.tp.dcis.data.operative.OperativeReport;
25  import de.paladinsinn.tp.dcis.data.operative.OperativeReportRepository;
26  import de.paladinsinn.tp.dcis.data.operative.OperativeRepository;
27  import de.paladinsinn.tp.dcis.ui.components.TorgNotification;
28  import lombok.AllArgsConstructor;
29  import org.slf4j.Logger;
30  import org.slf4j.LoggerFactory;
31  import org.springframework.dao.DataAccessException;
32  import org.springframework.stereotype.Service;
33  
34  import javax.transaction.Transactional;
35  import java.util.Arrays;
36  
37  /**
38   * RemoveOperativeFromMissionListener --
39   *
40   * @author klenkes74 {@literal <rlichti@kaiserpfalz-edv.de>}
41   * @since 0.1.0  2021-04-11
42   */
43  @Service
44  @AllArgsConstructor
45  public class RemoveOperativeFromMissionListener implements ComponentEventListener<RemoveOperativeFromMissionEvent> {
46      private static final Logger LOG = LoggerFactory.getLogger(RemoveOperativeFromMissionListener.class);
47  
48      private final MissionReportRepository missionReportRepository;
49      private final OperativeRepository operativeRepository;
50      private final OperativeReportRepository operativeReportRepository;
51  
52      @Override
53      @Transactional
54      public void onComponentEvent(RemoveOperativeFromMissionEvent event) {
55          OperativeReport report = event.getOperativeReport();
56  
57          MissionReport mission = report.getReport();
58          Operative operative = report.getOperative();
59  
60          String missionId = report.getReport().getId().toString();
61          String missionCode = report.getReport().getMission().getCode();
62          String operativeId = report.getOperative().getId().toString();
63          String operativeName = report.getOperative().getName();
64  
65          LOG.info("Deleting operative report. report={}, execution={}, mission='{}', operative='{}'",
66                  report.getId(), missionId, missionCode, operativeName);
67  
68          report.setOperative(null);
69          report.setMissionReport(null);
70  
71          try {
72              operativeRepository.save(operative);
73              missionReportRepository.save(mission);
74  
75              operativeReportRepository.delete(report);
76  
77              new TorgNotification(
78                      "missionreport.remove-operative.success",
79                      null,
80                      null,
81                      Arrays.asList(
82                              report.getId().toString(), missionId, operativeId,
83                              missionCode, operativeName
84                      )
85              ).open();
86          } catch (DataAccessException e) {
87              new TorgNotification(
88                      "missionreport.remove-operative.failed",
89                      null,
90                      null,
91                      Arrays.asList(
92                              report.getId().toString(), missionId, operativeId,
93                              missionCode, operativeName
94                      )
95              ).open();
96          }
97      }
98  }
