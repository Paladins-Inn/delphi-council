18  package de.paladinsinn.tp.dcis.ui.views.operativespecialreports;
19  
20  import com.sun.istack.NotNull;
21  import com.vaadin.flow.component.ComponentEvent;
22  import de.paladinsinn.tp.dcis.data.operative.Operative;
23  import de.paladinsinn.tp.dcis.data.specialmissions.SpecialMission;
24  import de.paladinsinn.tp.dcis.ui.components.TorgForm;
25  import de.paladinsinn.tp.dcis.ui.views.specialmissions.SpecialMissionForm;
26  import lombok.Getter;
27  
28  import java.util.StringJoiner;
29  
30  /**
31   * AddOperativeToSpecialMissionEvent -- Adds an {@link Operative} to a {@link SpecialMission}.
32   *
33   * @author klenkes74 {@literal <rlichti@kaiserpfalz-edv.de>}
34   * @since 0.3.0  2021-04-18
35   */
36  @Getter
37  public class AddOperativeToSpecialMissionEvent extends ComponentEvent<TorgForm> {
38  
39      /**
40       * The operative to be added.
41       */
42      private final Operative operative;
43  
44      /**
45       * The mission to be changed.
46       */
47      private final SpecialMission mission;
48  
49      /**
50       * Creates a new event using the given source and indicator whether the
51       * event originated from the client side or the server side.
52       *
53       * @param source    the source component
54       * @param mission   the mission the operative is added to
55       * @param operative the operative to add to the mission
56       */
57      public AddOperativeToSpecialMissionEvent(@NotNull final SpecialMissionForm source, @NotNull final SpecialMission mission, @NotNull final Operative operative) {
58          super(source, false);
59  
60          this.mission = mission;
61          this.operative = operative;
62      }
63  
64      @Override
65      public String toString() {
66          return new StringJoiner(", ", AddOperativeToSpecialMissionEvent.class.getSimpleName() + "[", "]")
67                  .add("operative=" + operative)
68                  .add("mission=" + mission)
69                  .toString();
70      }
71  }
