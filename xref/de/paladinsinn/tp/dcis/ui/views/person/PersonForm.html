131     public void init() {
132         if (initialized || data == null || user == null) {
133             LOG.debug("Already initialized or not initializable. initialized={}, data={}, user={}, locale={}",
134                     initialized, data, user, locale);
135             return;
136         }
137 
138         addListener(PersonSaveEvent.class, personSaveService);
139         user.allow(user.getPerson().getName().equals(data.getUsername()));
140         ensureLocale();
141 
142         LOG.debug("initializing. data={}, user={}, locale={}", data, user, locale);
143 
144         form = new FormLayout();
145         form.setResponsiveSteps(
146                 new FormLayout.ResponsiveStep("1px", 1),
147                 new FormLayout.ResponsiveStep("400px", 2),
148                 new FormLayout.ResponsiveStep("800px", 3)
149         );
150 
151         id = generateTextField(true, false);
152         username = generateTextField(!user.isAdmin(), true);
153         password = generatePasswordField(user.isReadonly(), false);
154         email = generateEmailField(user.isReadonly(), true);
155         name = generateTextField(user.isReadonly(), true);
156         lastName = generateTextField(user.isReadonly(), true);
157         firstName = generateTextField(user.isReadonly(), true);
158 
159         language = new I18nSelector(data.getLocale());
160         language.setReadOnly(user.isReadonly());
161         language.setRequiredIndicatorVisible(true);
162 
163         lastLogin = new DateTimePicker();
164         lastLogin.setReadOnly(true);
165 
166         deleted = new DateTimePicker();
167         deleted.setReadOnly(!user.isAdmin());
168 
169         lastPasswordChange = new DateTimePicker();
170         lastPasswordChange.setReadOnly(true);
171 
172         expiryDate = new DateTimePicker();
173         expiryDate.setReadOnly(!user.isAdmin());
174         expiryDate.setRequiredIndicatorVisible(true);
175 
176         status = new CheckboxGroup<>();
177         status.setReadOnly(!user.isAdmin() && !user.isOrga() && !user.isJudge());
178 
179         flags = new CheckboxGroup<>();
180         flags.setReadOnly(user.isReadonly());
181 
182         roles = new CheckboxGroup<>();
183         roles.setItems(RoleName.PERSON.getRoleNamesWithoutGM());
184         roles.setReadOnly(!user.isAdmin() && !user.isOrga());
185 
186         avatar = new Avatar(
187                 "person.avatar",
188                 data,
189                 50, 150,
190                 50, 150,
191                 16777215
192         );
193         FormLayout.FormItem avatarItem = form.addFormItem(avatar, getTranslation("person.avatar.caption"));
194 
195         actions = new TorgActionBar(
196                 "buttons",
197                 event -> { // save
198                     scrape();
199                     getEventBus().fireEvent(new PersonSaveEvent(this, false));
200                 },
201                 ev -> { // reset
202                     data = oldData;
203                     populate();
204                 },
205                 ev -> { // cancel
206                     getUI().ifPresent(ui -> ui.getPage().getHistory().back());
207                 },
208                 null
209         );
210         actions.setReadOnly(user.isReadonly()
211                 && !user.getPerson().equals(data)
212                 && !user.isAdmin()
213         );
214 
215 
216         form.add(username, password, status,
217                 email, flags,
218                 name, language,
219                 lastName, firstName, new Span(),
220                 roles, expiryDate,
221                 lastPasswordChange, lastLogin, deleted);
222 
223         actions.translate();
224         form.add(actions);
225 
226         form.setColspan(avatarItem, 3);
227         form.setColspan(email, 2);
228         form.setColspan(name, 2);
229         form.setColspan(roles, 2);
230         form.setColspan(actions, 3);
231         add(form);
232 
233         // mark as initialized.
234         initialized = true;
235     }

272     public void setData(@NotNull Person data) {
273         try {
274             this.oldData = data.clone();
275         } catch (CloneNotSupportedException e) {
276             // should not happen
277             LOG.error("All entities need to be clonable: data={}", data);
278         }
279         this.data = data;
280 
281         LOG.debug("Set data. id={}, name={}",
282                 this.data.getId(), this.data.getName());
283 
284         init();
285         populate();
286     }

288     private void populate() {
289         if (!initialized || data == null) {
290             LOG.warn("Form is not initialized or data is not set. initialized={}, data={}", initialized, data);
291             return;
292         }
293 
294         if (data.getId() != null) {
295             id.setValue(data.getId().toString());
296         }
297         username.setValue(data.getUsername());
298         email.setValue(data.getEmail());
299 
300         avatar.setValue(data);
301 
302         name.setValue(data.getName());
303         lastName.setValue(data.getLastname());
304         firstName.setValue(data.getFirstname());
305 
306         lastLogin.setValue(data.getStatus().getLastLogin().toLocalDateTime());
307         expiryDate.setValue(data.getStatus().getExpiry().toLocalDateTime());
308         lastPasswordChange.setValue(data.getStatus().getCredentialsChange().toLocalDateTime());
309 
310         roles.setValue(RoleName.PERSON.getActiveRoleNames(data));
311 
312         if (roles.getValue().contains(RoleName.GM.name())) {
313             flags.add(getTranslation("person.status-is-gm.caption"));
314         }
315         if (data.isGravatarAllowed()) {
316             flags.add(getTranslation("person.status-allow-gravatar.caption"));
317         }
318         flags.addSelectionListener(e -> {
319             roles.getValue().remove(RoleName.GM.name());
320 
321             for (String s : e.getAllSelectedItems()) {
322                 if (s.equals(getTranslation("person.status-is-gm.caption"))) {
323                     roles.getValue().add(RoleName.GM.name());
324                 }
325             }
326         });
327 
328         language.setValue(data.getLocale());
329     }
330 
331     private void scrape() {
332         if (data == null) data = new Person();
333 
334         data.setUsername(username.getValue());
335 
336         if (!password.isEmpty()) {
337             LOG.info("Changing password. user='{}'", data.getUsername());
338             data.setPassword(password.getValue());
339         }
340 
341         data.setEmail(email.getValue());
342 
343         data.setName(name.getValue());
344         LOG.info("Changing name. name='{}'", data.getName());
345 
346         data.setLastname(lastName.getValue());
347         data.setFirstname(firstName.getValue());
348 
349         data.getStatus().setLocked(false);
350         data.getStatus().setEnabled(true);
351         for (String b : status.getValue()) {
352             if (getTranslation("person.status-locked.caption").equals(b)) {
353                 data.getStatus().setLocked(true);
354             } else if (getTranslation("person.status-enabled.caption").equals(b)) {
355                 data.getStatus().setEnabled(true);
356             }
357         }
358         data.getStatus().setLastLogin(lastLogin.getValue().atOffset(UTC));
359         data.getStatus().setCredentialsChange(lastPasswordChange.getValue().atOffset(UTC));
360         data.getStatus().setExpiry(expiryDate.getValue().atOffset(UTC));
361 
362         data.getRoles().clear();
363         for (String b : roles.getValue()) {
364             data.getRoles().add(new Role(RoleName.valueOf(b)));
365         }
366 
367         data.disableGravatar();
368 
369         for (String b : flags.getValue()) {
370             if (getTranslation("person.status-allow-gravatar.caption").equals(b)) {
371                 data.enableGravatar();
372             }
373         }
374 
375         if (deleted.getValue() != null) {
376             data.getStatus().setDeleted(deleted.getValue().atOffset(UTC));
377         }
378 
379         data.setLocale(language.getValue());
380     }

387     @Override
388     public void translate() {
389         if (locale == null) {
390             setLocale(VaadinSession.getCurrent().getLocale());
391         }
392 
393         // Form fields
394         LOG.trace("Adding all form elements.");
395         id.setLabel(getTranslation("input.id.caption"));
396         id.setHelperText(getTranslation("input.id.help"));
397 
398         name.setLabel(getTranslation("person.name.caption"));
399         name.setHelperText(getTranslation("person.name.help"));
400 
401         email.setLabel(getTranslation("person.email.caption"));
402         email.setHelperText(getTranslation("person.email.help"));
403 
404         username.setLabel(getTranslation("person.username.caption"));
405         username.setHelperText(getTranslation("person.username.help"));
406         password.setLabel(getTranslation("person.password.caption"));
407         password.setHelperText(getTranslation("person.password.help"));
408 
409         roles.setLabel(getTranslation("person.roles.caption"));
410         roles.setHelperText(getTranslation("person.roles.help"));
411 
412         String statusEnabled = getTranslation("person.status-enabled.caption");
413         String statusLocked = getTranslation("person.status-locked.caption");
414         HashSet<String> statusOptionsSelected = new HashSet<>(2);
415         if (data.isEnabled()) {
416             statusOptionsSelected.add(getTranslation("person.status-enabled.caption"));
417         }
418         if (!data.isAccountNonLocked()) {
419             statusOptionsSelected.add(getTranslation("person.status-locked.caption"));
420         }
421         status.removeAll();
422         status.setLabel(getTranslation("person.status.caption"));
423         status.setHelperText(getTranslation("person.status.help"));
424         status.setItems(Arrays.asList(statusEnabled, statusLocked));
425         status.setValue(statusOptionsSelected);
426 
427         final String userChangeableFlagIsGm = getTranslation("person.status-is-gm.caption");
428         final String userChangeableFlagAllowGravatar = getTranslation("person.status-allow-gravatar.caption");
429         final HashSet<String> userChangeableFlagSelected = new HashSet<>(2);
430         if (data.isGravatarAllowed()) {
431             userChangeableFlagSelected.add(getTranslation("person.status-allow-gravatar.caption"));
432         }
433         if (roles.getValue().contains(RoleName.GM.name())) {
434             userChangeableFlagSelected.add(getTranslation("person.status-is-gm.caption"));
435         }
436         flags.removeAll();
437         flags.setLabel(getTranslation("person.user-changeable-flags.caption"));
438         flags.setItems(Arrays.asList(userChangeableFlagIsGm, userChangeableFlagAllowGravatar));
439         flags.setValue(userChangeableFlagSelected);
440 
441         lastName.setLabel(getTranslation("person.last-name.caption"));
442         lastName.setHelperText(getTranslation("person.last-name.help"));
443         firstName.setLabel(getTranslation("person.first-name.caption"));
444         firstName.setHelperText(getTranslation("person.first-name.help"));
445         deleted.setLabel(getTranslation("person.deleted.caption"));
446         deleted.setHelperText(getTranslation("person.deleted.help"));
447         expiryDate.setLabel(getTranslation("person.expiry-date.caption"));
448         expiryDate.setHelperText(getTranslation("person.expiry-date.help"));
449         lastPasswordChange.setLabel(getTranslation("person.last-password-change.caption"));
450         lastPasswordChange.setHelperText(getTranslation("person.last-password-change.help"));
451         lastLogin.setLabel(getTranslation("person.last-login.caption"));
452         lastLogin.setHelperText(getTranslation("person.last-login.help"));
453 
454         actions.translate();
455     }

463     @Override
464     public void setLocale(Locale locale) {
465         if (locale == null || (locale.equals(this.locale))) {
466             LOG.debug("Locale already set or new locale would be null, old={}, new={}", this.locale, locale);
467             return;
468         }
469 
470         this.locale = locale;
471         translate();
472     }

474     private String getTranslation(@NotNull final String key) {
475         try {
476             return super.getTranslation(key);
477         } catch (NullPointerException e) {
478             LOG.warn("Can't call translator from vaadin: {}", e.getMessage());
479             return "!" + key;
480         }
481     }
