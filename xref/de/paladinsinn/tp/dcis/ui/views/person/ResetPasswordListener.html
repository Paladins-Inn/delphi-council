18  package de.paladinsinn.tp.dcis.ui.views.person;

20  import com.vaadin.flow.component.ComponentEventListener;
21  import de.paladinsinn.tp.dcis.data.person.PasswordResetToken;
22  import de.paladinsinn.tp.dcis.data.person.PasswordResetTokenRepository;
23  import de.paladinsinn.tp.dcis.data.person.Person;
24  import de.paladinsinn.tp.dcis.ui.components.TorgNotification;
25  import de.paladinsinn.tp.dcis.ui.views.login.LoginView;
26  import lombok.AllArgsConstructor;
27  import org.slf4j.Logger;
28  import org.slf4j.LoggerFactory;
29  import org.springframework.stereotype.Service;
30  import org.springframework.transaction.annotation.Transactional;
31  
32  import javax.validation.constraints.NotNull;
33  import java.util.Collections;
34  import java.util.List;
35  import java.util.Optional;
36  import java.util.UUID;
37  
38  /**
39   * ResetPasswordListener -- Resets the password for a given user with the provided new password.
40   *
41   * @author klenkes74 {@literal <rlichti@kaiserpfalz-edv.de>}
42   * @since 1.1.0  2021-05-07
43   */
44  @Service
45  @AllArgsConstructor
46  public class ResetPasswordListener implements ComponentEventListener<ResetPasswordEvent> {
47      private static final Logger LOG = LoggerFactory.getLogger(ResetPasswordListener.class);
48  
49      private final PasswordResetTokenRepository tokenRepository;
50  
51      @Transactional
52      @Override
53      public void onComponentEvent(@NotNull final ResetPasswordEvent event) {
54          LOG.info("User want's to reset his password. event={}", event);
55  
56          Optional<Person> person = loadByToken(event.getToken());
57  
58          person.ifPresentOrElse(
59                  p -> {
60                      p.setPassword(event.getPassword());
61                      removeAllPersonalToken(p);
62  
63                      new TorgNotification(
64                              "password-reset.changed-password.success",
65                              null,
66                              null,
67                              Collections.singletonList(p.getUsername())
68                      ).open();
69                  },
70                  () -> new TorgNotification(
71                          "password-reset.invalid-token",
72                          null,
73                          null,
74                          Collections.emptyList()
75                  ).open()
76          );
77  
78          event.getSource().getUI().ifPresent(ui -> ui.navigate(LoginView.ROUTE));
79      }
80  
81  
82      private Optional<Person> loadByToken(@NotNull final UUID tokenId) {
83          Optional<PasswordResetToken> token = tokenRepository.findById(tokenId);
84  
85          Optional<Person> result = Optional.empty();
86          return token.map(PasswordResetToken::getPerson).or(() -> result);
87      }
88  
89      private void removeAllPersonalToken(@NotNull final Person person) {
90          List<PasswordResetToken> tokens = tokenRepository.findAllByPerson(person);
91  
92          tokens.forEach(tokenRepository::delete);
93      }
94  }
