42  
43  import javax.annotation.PostConstruct;

46  import java.util.Locale;

48  import java.util.UUID;

50  /**
51   * SpecialMissionEditorView -- Editor for adding new {@link SpecialMission} or editing existing ones.
52   *
53   * @author klenkes74 {@literal <rlichti@kaiserpfalz-edv.de>}
54   * @since 0.1.0  2021-04-06
55   */
56  @Route(value = "specialmission/:id?/:gm?", layout = MainView.class)
57  @I18nPageTitle("specialmission.editor.caption")
58  @CssImport("./views/edit-view.css")
59  @Secured({"JUDGE", "ORGA", "ADMIN", "GM"})
60  public class SpecialMissionEditorView extends Div implements BeforeEnterObserver, LocaleChangeObserver, TranslatableComponent, Serializable, AutoCloseable {
61      private static final Logger LOG = LoggerFactory.getLogger(SpecialMissionEditorView.class);
62  
63      @Autowired
64      private SpecialMissionForm form;
65  
66      @Autowired
67      private SpecialMissionRepository missionRepository;
68  
69      @Autowired
70      private PersonRepository personRepository;
71  
72      @Autowired
73      private LoggedInUser user;
74  
75  
76      private Locale locale;
77  
78      @PostConstruct
79      public void init() {
80          setLocale(VaadinSession.getCurrent().getLocale());
81          LOG.debug("Loading form. locale={}", locale);
82  
83          addClassName("edit-view");
84  
85          add(form);
86          translate();
87      }
88  
89  
90      @Override
91      public void beforeEnter(BeforeEnterEvent event) {
92          Optional<String> id = event.getRouteParameters().get("id");
93          Optional<String> gm = event.getRouteParameters().get("gm");
94          LOG.trace("Entering form. id={}", id);
95  
96          id.ifPresentOrElse(
97                  e -> missionRepository.findById(UUID.fromString(e)).ifPresentOrElse(
98                          this::setData,
99                          () -> generateNewSpecialMission(UUID.fromString(e), gm)
100                 ),
101                 () -> generateNewSpecialMission(UUID.randomUUID(), gm)
102         );
103     }
104 
105     private void generateNewSpecialMission(
106             UUID missionId,
107             @SuppressWarnings("OptionalUsedAsFieldOrParameterType") Optional<String> gameMasterId
108     ) {
109         gameMasterId.ifPresentOrElse(
110                 gmid -> {
111                     Optional<Person> person = personRepository.findById(UUID.fromString(gmid));
112                     person.ifPresentOrElse(
113                             p -> {
114                                 SpecialMission newMission = new SpecialMission();
115                                 newMission.setGameMaster(p);
116                                 newMission.setCode(missionId);
117                                 setData(newMission);
118                             },
119                             this::returnWithErrorMessage
120                     );
121                 },
122                 () -> {
123                     SpecialMission newMission = new SpecialMission();
124                     newMission.setGameMaster(user.getPerson());
125                     newMission.setCode(missionId);
126                     setData(newMission);
127                 }
128 
129         );
130     }
131 
132     private void returnWithErrorMessage() {
133         new TorgNotification(
134                 "specialmission.missing-param.gm",
135                 null,
136                 new ArrayList<>(),
137                 new ArrayList<>()
138         ).open();
139         getUI().ifPresent(ui -> ui.getPage().getHistory().back());
140     }
141 
142 
143     private void setData(@NotNull SpecialMission data) {
144         if (data == null || data.equals(form.getData().orElse(null))) {
145             LOG.info("Data is null or data didn't change - will do nothing. data={}", data);
146             return;
147         }
148         LOG.trace("Loaded mission. mission={}", data);
149 
150         form.setData(data);
151         translate();
152     }
153 
154 
155     @Override
156     public void localeChange(LocaleChangeEvent event) {
157         LOG.trace("Locale change event. locale={}", event.getLocale());
158 
159         setLocale(event.getLocale());
160     }
161 
162     @Override
163     public void translate() {
164         if (locale == null || form.getData().isEmpty()) {
165             LOG.debug("No locale or no data - will do nothing. locale={}, data={}", locale, form.getData());
166             return;
167         }
168 
169         form.translate();
170     }
171 
172     @Override
173     public void setLocale(Locale locale) {
174         if (this.locale != null && this.locale.equals(locale)) {
175             LOG.debug("Locale didn't change - will do nothing. locale={}", locale);
176             return;
177         }
178 
179         this.locale = locale;
180 
181         form.setLocale(locale);
182         translate();
183     }
184 
185     @Override
186     public void close() throws Exception {
187         LOG.trace("Closing form.");
188 
189         form.close();
190 
191         removeAll();
192 
193         LOG.debug("Closed form.");
194     }
195 }
