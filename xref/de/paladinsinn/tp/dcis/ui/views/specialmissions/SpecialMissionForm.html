48  /**
49   * SpecialMissionForm -- Edits/displays the data for a mission.
50   *
51   * @author klenkes74 {@literal <rlichti@kaiserpfalz-edv.de>}
52   * @since 0.3.0  2021-04-18
53   */
54  @Service
55  @Scope(SCOPE_PROTOTYPE)
56  public class SpecialMissionForm extends TorgForm<SpecialMission> {
57      private static final Logger LOG = LoggerFactory.getLogger(SpecialMissionForm.class);
58      public static final int DEFAULT_PAYMENT = 250;
59      public static final int DEFAULT_XP = 5;
60  
61      /**
62       * service for writing data to.
63       */
64      private final SpecialMissionService missionService;
65      private final OperativeRepository operativeRepository;
66  
67      private final AddOperativeToSpecialMissionListener addOperativeToMissionListener;
68  
69  
70      private final TextField title = new TextField();
71  
72      private final TextField id = new TextField();
73      private final TextField code = new TextField();
74      private final ComboBox<Clearance> clearance = new ComboBox<>();
75      private final DatePicker missionDate = new I18nDatePicker();
76      private final TextField publication = new TextField();
77      private final TextField image = new TextField();
78      private final TextArea description = new TextArea();
79      private final IntegerField payment = new IntegerField();
80      private final IntegerField xp = new IntegerField();
81  
82      private OperativeSelector operatives;
83  
84  
85      @Autowired
86      public SpecialMissionForm(
87              @NotNull final SpecialMissionService missionService,
88              @NotNull final AddOperativeToSpecialMissionListener addOperativeToMissionListener,
89              @NotNull final OperativeRepository operativeRepository,
90              @NotNull final LoggedInUser user
91      ) {
92          super(user);
93  
94          this.missionService = missionService;
95          this.operativeRepository = operativeRepository;
96          this.addOperativeToMissionListener = addOperativeToMissionListener;
97      }
98  
99      protected void init() {
100         if (data == null || user == null || initialized) {
101             LOG.debug(
102                     "Can't initialize or already initialized. initialized={}, data={}, user={}",
103                     initialized, data, user
104             );
105             return;
106         }
107 
108         super.init();
109 
110         addListener(SpecialMissionSaveEvent.class, missionService);
111         addListener(AddOperativeToSpecialMissionEvent.class, addOperativeToMissionListener);
112 
113         actions = new TorgActionBar(
114                 "buttons",
115                 event -> { // save
116                     scrape();
117                     getEventBus().fireEvent(new SpecialMissionSaveEvent(this, data));
118 
119                     for (Operative o : operatives.getValue()) {
120                         fireEvent(new AddOperativeToSpecialMissionEvent(this, data, o));
121                     }
122                 },
123                 event -> { // reset
124                     LOG.info("Resetting data from: displayed={}, new={}", data, oldData);
125                     resetData();
126                 },
127                 event -> { // cancel
128                     getUI().ifPresent(ui -> ui.getPage().getHistory().back());
129                 },
130                 null
131         );
132         actions.setReadOnly(
133                 user.isReadonly()
134                         && !user.getPerson().equals(data.getGameMaster())
135                         && !user.isAdmin()
136                         && !user.isOrga()
137                         && !user.isJudge()
138         );
139 
140         id.setReadOnly(true);
141 
142         code.setReadOnly(true);
143 
144         title.setRequired(true);
145         title.setReadOnly(user.isReadonly());
146 
147         clearance.setDataProvider(Clearance.ANY.dataProvider());
148         clearance.setAllowCustomValue(false);
149         clearance.setRequired(true);
150         clearance.setReadOnly(user.isReadonly());
151 
152         publication.setRequired(false);
153         publication.setReadOnly(user.isReadonly());
154 
155         image.setRequired(false);
156         image.setReadOnly(user.isReadonly());
157 
158         description.addClassName("long-text");
159         description.setRequired(true);
160         description.setReadOnly(user.isReadonly());
161 
162         payment.setMin(100);
163         payment.setMax(1000);
164         payment.setStep(50);
165         payment.setValue(DEFAULT_PAYMENT);
166         payment.setReadOnly(user.isReadonly());
167 
168         xp.setMin(1);
169         xp.setMax(50);
170         xp.setStep(1);
171         xp.setValue(DEFAULT_XP);
172         xp.setReadOnly(user.isReadonly());
173 
174         operatives = new OperativeSelector("missionreport.add-operatives", operativeRepository);
175         operatives.setReadonly(user.isReadonly());
176         operatives.setVisible(!user.isReadonly());
177 
178         LOG.trace("adding all form elements.");
179         form.add(
180                 code, clearance, missionDate,
181                 title,
182                 publication,
183                 payment, xp, image,
184                 description,
185                 operatives);
186 
187         form.add(actions);
188         form.setColspan(actions, 3);
189 
190 
191         form.setColspan(title, 3);
192         form.setColspan(publication, 3);
193         form.setColspan(description, 3);
194 
195         getContent().add(form);
196 
197         initialized = true;
198     }
199 
200 
201     public void populate() {
202         if (data == null) {
203             LOG.warn("Tried to populate form data without a mission defined.");
204             return;
205         }
206 
207         if (data.getId() != null) {
208             id.setValue(data.getId().toString());
209         }
210 
211         if (data.getCode() != null) {
212             code.setValue(data.getCode().toString());
213         } else {
214             code.setValue(UUID.randomUUID().toString());
215         }
216 
217         if (data.getGameMaster() != null) {
218             data.setGameMaster(user.getPerson());
219         }
220 
221         bindData(data.getMissionDate(), missionDate);
222 
223         bindData(data.getTitle(), title);
224         bindData(data.getClearance(), clearance);
225         bindData(data.getPublication(), publication);
226         bindData(data.getImageUrl(), image);
227         bindData(data.getDescription(), description);
228         bindData(data.getPayment(), payment);
229         bindData(data.getXp(), xp);
230     }
231 
232     protected void scrape() {
233         if (data == null) data = new SpecialMission();
234 
235         data.setCode(UUID.fromString(code.getValue()));
236         data.setTitle(title.getValue());
237         data.setClearance(clearance.getValue());
238         data.setMissionDate(missionDate.getValue());
239         data.setPublication(publication.getValue());
240         data.setImageUrl(image.getValue());
241         data.setDescription(description.getValue());
242         data.setPayment(payment.getValue());
243         data.setXp(xp.getValue());
244         data.preUpdate();
245     }
246 
247     @Override
248     public void translate() {
249         if (user == null || data == null || locale == null) {
250             LOG.warn(
251                     "Can't build special mission form. user={}, mission={}, locale={}",
252                     user, data, locale
253             );
254 
255             return;
256         }
257 
258         LOG.debug("Building special mission edit form. locale={}, mission={}", locale, data);
259 
260         // Form fields
261         LOG.trace("Translating form elements.");
262         id.setLabel(getTranslation("input.id.caption"));
263         id.setHelperText(getTranslation("input.id.help"));
264 
265         code.setLabel(getTranslation("mission.code.caption"));
266         code.setHelperText(getTranslation("mission.code.help"));
267 
268         title.setLabel(getTranslation("mission.title.caption"));
269         title.setHelperText(getTranslation("mission.title.help"));
270 
271         clearance.setLabel(getTranslation("torg.clearance.caption"));
272         clearance.setHelperText(getTranslation("torg.clearance.help"));
273         clearance.setItemLabelGenerator((ItemLabelGenerator<Clearance>) item -> getTranslation("torg.clearance." + item.name()));
274 
275         missionDate.setLabel(getTranslation("specialmission.date.caption"));
276         missionDate.setHelperText(getTranslation("specialmission.date.help"));
277 
278         publication.setLabel(getTranslation("mission.publication.caption"));
279         publication.setHelperText(getTranslation("mission.publication.help"));
280 
281         image.setLabel(getTranslation("mission.image.caption"));
282         image.setHelperText(getTranslation("mission.image.help"));
283 
284         description.setLabel(getTranslation("mission.description.caption"));
285         description.setHelperText(getTranslation("mission.description.help"));
286 
287         payment.setLabel(getTranslation("mission.payment.caption"));
288         payment.setHelperText(getTranslation("mission.description.help"));
289 
290         xp.setLabel(getTranslation("mission.xp.caption"));
291         xp.setHeight(getTranslation("mission.xp.help"));
292     }
